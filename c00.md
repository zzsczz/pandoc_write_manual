  - "第"
  - ""
  - ""

figPrefix: "图"

tblPrefix: "表"

secPrefix: "小节"
listingTitle: "语法代码"
secHeaderSuffix:
  - "章"
  - "节"
  - ""
sectionsDepth: -1
numberSections: True
refIndexTemplate:
    sec: $$i$$$$suf$$ ($$t$$)
    default: $$i$$$$suf$$

eqnBlockInlineMath: true
equationNumberTeX: \\tag
eqnIndexTemplate: ($$i$$) #这个是给编号加上括号
eqnPrefixTemplate: 式($$i$$) #给引用的公式编号加上括号
